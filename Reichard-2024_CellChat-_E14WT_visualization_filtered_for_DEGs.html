library(CellChat)
library(patchwork)
options(stringsAsFactors = FALSE)

cellchat_dir <- "data/cellchat/"
plots_dir <- "plots/Reichard_et_al/"

cellchat.WT14 <- readRDS("data/cellchat/ZB/02_E14WT_CellChat_with_networks.rds")

Load the CellChat objects

df.net_from_IN <- subsetCommunication(cellchat.WT14, slot.name = "net",
sources.use = c(19), targets.use = c(1, 2, 3, 4, 5))


from_IN <- c("Thbs1", "Reln", "Ptprm", "Nrxn3", "Nrxn2", "Ncam1", "Negr1", "Lrrc4b", "Gad2", "Slc6a1", "Slc32a1", "Gad1", "Efnb2", "Sst", "Sema3a", "Ptn", "Mdk", "Tenm3", "Slit2", "Sema5a", "Cntn2")

subset_from_IN <- df.net_from_IN[(df.net_from_IN$receptor %in% from_IN) | (df.net_from_IN$ligand %in% from_IN), ]
LR_from_IN <- as.data.frame(subset_from_IN$interaction_name)
colnames(LR_from_IN) <- "interaction_name"

Visualize cell-cell communication mediated by multiple ligand-receptors or signaling pathways

Communication from interneurons to glutamatergic populations

Bubble plot Sst+ MGE interneurons to AP

Only the DEGs upon Dnmt1 KO are present

netVisual_bubble(cellchat.WT14, 
                 sources.use = c(19), 
                 targets.use = c(1, 2, 3), remove.isolate = TRUE,
                 pairLR.use = LR_from_IN)

ggsave(filename = paste0(plots_dir, "E14_IN_to_AP_Dnmt1_DEGs.pdf"),
